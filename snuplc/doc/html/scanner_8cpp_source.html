162 string CToken::escape(EToken type, const string text)
163 {
164  // inverse of CToken::unescape()
165 
166  const char *t = text.c_str();
167  string s;
168 
169  while ((type == tCharConst) || (*t != '\0')) {
170  char c = *t;
171 
172  switch (c) {
173  case '\n': s += "\\n"; break;
174  case '\t': s += "\\t"; break;
175  case '\0': s += "\\0"; break;
176  case '\'': if (type == tCharConst) s += "\\'";
177  else s += c;
178  break;
179  case '\"': if (type == tStringConst) s += "\\\"";
180  else s += c;
181  break;
182  case '\\': s += "\\\\"; break;
183  default : if ((c < ' ') || (c == '\x7f')) {
184  // ASCII characters 0x80~0xff satisfy (signed char c < ' ')
185  char str[5];
186  snprintf(str, sizeof(str), "\\x%02x", (unsigned char)c);
187  s += str;
188  } else {
189  s += c;
190  }
191  }
192  if (type == tCharConst) break;
193  t++;
194  }
195 
196  return s;
197 }
