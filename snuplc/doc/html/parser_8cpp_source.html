150 CAstStatement* CParser::statSequence(CAstScope *s)
151 {
152  //
153  // statSequence ::= [ statement { ";" statement } ].
154  // statement ::= assignment.
155  //
156  // FIRST(statSequence) = { tLetter }
157  // FOLLOW(statSequence) = { tDot }
158  //
159  // FIRST(statement) = { tLetter }
160  // FOLLOW(statement) = { tSemicolon, tDot }
161  //
162 
163  // The linking of statement sequences is a bit akward here because
164  // we implement statSequence as a loop and not recursively.
165  // We keep a 'head' that points to the first statement and is finally
166  // returned at the end of the function. Head can be NULL if no statement
167  // is present.
168  // In the loop, we track the end of the linked list using 'tail' and
169  // attach new statements to that tail.
170  CAstStatement *head = NULL;
171 
172  if (_scanner->Peek().GetType() != tDot) {
173  CAstStatement *tail = NULL;
174 
175  do {
176  CAstStatement *st = NULL;
177 
178  switch (_scanner->Peek().GetType()) {
179  // statement ::= assignment
180  case tLetter:
181  st = assignment(s);
182  break;
183 
184  default:
185  SetError(_scanner->Peek(), "statement expected.");
186  break;
187  }
188 
189  assert(st != NULL);
190  if (head == NULL) head = st;
191  else tail->SetNext(st);
192  tail = st;
193 
194  if (_scanner->Peek().GetType() == tDot) break;
195 
196  Consume(tSemicolon);
197  } while (!_abort);
198  }
199 
200  return head;
201 }
